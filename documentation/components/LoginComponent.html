import {Component, Inject, Input, OnInit} from '@angular/core';

import { UserDetails } from '@ukis/datatypes/UserDetails'
import {Observable} from 'rxjs/Observable';
import {BehaviorSubject} from 'rxjs/BehaviorSubject';
import {UserService} from '@ukis/services/src/app/user/user.service';
import {Subscription} from 'rxjs/Subscription';
import {FormBuilder, FormControl, FormGroup, Validators} from '@angular/forms';

@Component({
  selector: 'ukis-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit {
  @Input('authService') authService: any;

  usrSubsription: Subscription;
  user: UserDetails;

 // usrInfoForm: FormGroup;

  usrInfoForm = new FormGroup({
    usrName: new FormControl('', Validators.required),
    usrPass: new FormControl('', Validators.required),
  });

  loginName: string;
  pass: string;

  constructor(@Inject(UserService)private usrSvc: UserService, fb: FormBuilder) {
    this.usrSubsription = this.usrSvc.getUser().subscribe(usr => { this.user = usr;});

  }

  ngOnInit() {
  }

  login() {
    this.authService.login(this.usrInfoForm.get("usrName").value, this.usrInfoForm.get("usrPass").value).subscribe(usr => {
      this.usrSvc.setUser(usr);
      console.log('you are logged in as: ', this.user.firstName);
    });
  }

  logout() {
    this.usrSvc.setUser(this.authService.logout());
    console.log('logedout: ', this.user.firstName);
  }

}
