import { Component, OnInit, Inject, ViewEncapsulation, OnDestroy, AfterViewChecked, Input } from '@angular/core';

import { MapState } from '@ukis/datatypes/MapState';
import { MapstateService } from '@ukis/services/src/app/mapstate/mapstate.service';
import { Subscription } from 'rxjs/Subscription';
import { Ol4MapSvc, IOl4MapSvc } from './ol-map.service';
import { LayersService } from '@ukis/services/src/app/layers/layers.service';
import { ILayer, IRasterLayer, IVectorLayer } from '@ukis/datatypes';


import * as ol from 'openlayers'
import { Observable, BehaviorSubject } from 'rxjs';
import { ArgumentType } from '@angular/compiler/src/core';
import { Subject } from 'rxjs/Subject';
import 'rxjs/add/operator/mergeMap'

@Component({
  selector: 'ukis-ol-map',
  templateUrl: './ol-map.component.html',
  styleUrls: ['./ol-map.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class OlMapComponent implements OnInit, OnDestroy {
  @Input('store') store: any; //Array of GeoJSON fetures
  @Input('layers') layers: LayersService;

  map: ol.Map;
  view: ol.View;

  viewSubj = new BehaviorSubject(this.view);

  zoom: number; //default value
  center: any //default value

  mapState: MapState;
  mapStateOn: Subscription;
  layersAddRemoveOn: Subscription;
  layersPropOn: Subscription;
  baselayersAddRemoveOn: Subscription;
  baselayersPropOn: Subscription;
  mapOn: ol.EventsListenerFunctionType;

  constructor( @Inject(MapstateService) private mapStateSvc: MapstateService, @Inject(Ol4MapSvc) private mapSvc: Ol4MapSvc) {
    this.zoom = 3;
    this.center = {
      lat: 0,
      lon: 0
    };
    //console.log(this.mapSvc)
    let ms = new MapState(this.zoom, this.center);
    this.mapState = ms;
    this.mapStateSvc.setMapState(ms);
    ///define initial center and zoom

  }

 

  ngOnInit() {
    this.initMap();
  }

  /** is  */
  ngAfterViewChecked() {
    this.map.updateSize();
  }

  ngOnDestroy() {
    this.mapStateOn.unsubscribe();
    this.layersAddRemoveOn.unsubscribe();
    this.layersPropOn.unsubscribe();
    this.baselayersAddRemoveOn.unsubscribe();
    this.baselayersPropOn.unsubscribe();
    this.map.un('moveend', this.mapOn)
  }


  private initMap() {
    var _EPSG = this.mapSvc.EPSG;

    var _baselayers = [];
    var _baselayerGroup = new ol.layer.Group(<any>{
      title: 'Base maps',
      type: 'baselayers',
      layers: _baselayers
    })

    //---------------------------------------------------------------------------------------------------
    //var _overlays = this.store.overlays;
    var _overlays = [];

    var _overlayGroup = new ol.layer.Group(<any>{
      title: 'Overlays',
      type: 'overlays',
      layers: _overlays
    })

    var _view = new ol.View({
      center: ol.proj.transform([this.center.lon, this.center.lat], 'EPSG:4326', _EPSG),
      zoom: this.zoom,
      projection: _EPSG
    })

    //controls
    /*
    var _fullscreen = new ol.control.FullScreen({ className: 'custom-full-screen' });
    var scaleLineControl = new ol.control.ScaleLine();
    var attribution = new ol.control.Attribution({
      //collapsible: false,
      collapsed: false
    });
    */

    var _map = new ol.Map({
      layers: [_baselayerGroup, _overlayGroup],
      view: _view,
      controls: [],
      //target: _target
      target: 'map',
    });

    this.map = _map;
    this.mapSvc.map = _map;
    this.view = this.map.getView();


    //add Control only if this functions is defined
    /*
    _map.addControl(attribution)
    _map.addControl(scaleLineControl)
    */


    //add events
    /*
    _map.on("click", (evt) => {
      this.$rootScope.$emit('ol.map.click', evt);
      this.markerLayerOnClick(evt, _map);
    })

    _map.on("moveend", (evt) => {
      console.log(evt)
      this.$rootScope.$emit('ol.map.change.view', evt);
    })

    _map.on('pointermove', (evt) => {
      this.$rootScope.$emit('ol.map.mousemove', evt);
      this.showMousePointerOnFeature(evt, _map);
    })
    */
    //----------------------
    //add/remove layers
    this.baselayersAddRemoveOn = this.layers.getBaseLayers().subscribe(layers => {
      if (layers.length != this.mapSvc.getLayers('overlays').length) {
        this.mapSvc.setBaseLayers(layers)
      }
    });

    //change layer visibility
    this.baselayersPropOn = this.layers.getBaseLayers().mergeMap(layers => layers).subscribe(layer => {
      let bllayer = this.mapSvc.getLayerByKey({ key: 'id', value: layer.ukisID }, 'baselayers');
      //console.log(layer, ollayer, this.mapSvc.map.getLayers())
      if (bllayer) {
        bllayer.setVisible(layer.visible);
        layer.visible != layer.visible;
      }
    });


    //add/remove layers
    this.layersAddRemoveOn = this.layers.getOverlays().subscribe(layers => {
      if (layers.length != this.mapSvc.getLayers('overlays').length) {
        this.mapSvc.setOverlays(layers)
      }
    });

    //change layer visibility and opacity
    this.layersPropOn = this.layers.getOverlays().mergeMap(layers => layers).subscribe(layer => {
      let ollayer = this.mapSvc.getLayerByKey({ key: 'id', value: layer.ukisID }, 'overlays');
      if (ollayer) {
        ollayer.setVisible(layer.visible);
        ollayer.setOpacity(layer.opacity);
        layer.opacity != layer.opacity;
        layer.visible != layer.visible;
      }
    });


    this.mapStateOn = this.mapStateSvc.getMapState().subscribe(mapState => {
      this.mapState = mapState;
      //console.log(mapState)
      this.mapSvc.setZoom(mapState.zoom, mapState.options.notifier);
      this.mapSvc.setCenter([mapState.center.lon, mapState.center.lat]);
    });


    this.mapOn = (evt) => {
      //console.log(this.mapState.zoom,this.mapState.center, this.mapState.options)
      let zoom = Math.round(this.view.getZoom());
      let center = {
        lat: this.view.getCenter()[1],
        lon: this.view.getCenter()[0]
      };
      let ms = new MapState(zoom, center, { notifier: 'map' });
      this.mapState = ms;
      this.mapStateSvc.setMapState(ms);
    }
    this.map.on('moveend', this.mapOn);




  }
}
