import {Component, Inject, OnDestroy, OnInit} from '@angular/core';

import { MapState } from '@ukis/datatypes/MapState';
import { MapstateService } from '@ukis/services/src/app/mapstate/mapstate.service';
import {Subscription} from 'rxjs/Subscription';

@Component({
  selector: 'ukis-mapnavigator',
  templateUrl: './mapnavigator.component.html',
  styleUrls: ['./mapnavigator.component.scss']
})
export class MapnavigatorComponent implements OnInit {

  mapState: MapState;
  subscription: Subscription;

  constructor(@Inject(MapstateService)private mapSvc: MapstateService) {
    this.subscription = this.mapSvc.getMapState().subscribe(mapState => { this.mapState = mapState;});
  }

  ngOnInit() {

  }

  ngOnDestroy() {
    this.subscription.unsubscribe();
  }

  zoomIn() {
    this.mapState.zoom++;
    this.setNewState(this.mapState);
  }
  zoomOut() {
    this.mapState.zoom--;
    this.setNewState(this.mapState);
  }

  setNewState(newState: MapState) {
    newState.options.notifier = 'user';
    this.mapSvc.setMapState(newState);
  }

  stateChange(ev) {

    this.setNewState(this.mapState);

  }
}
