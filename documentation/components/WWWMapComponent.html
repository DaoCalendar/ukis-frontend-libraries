import {
  Component, OnInit, Inject, ViewEncapsulation, OnDestroy, OnChanges, AfterViewInit, Input, ElementRef,
  ViewChild
} from '@angular/core';

import { MapState } from '@ukis/datatypes/MapState';
import { MapstateService } from '@ukis/services/src/app/mapstate/mapstate.service';
import { Subscription } from 'rxjs/Subscription';
import { WWWMapSvc, IWWWMapSvc } from './www-map.service';
import { LayersService } from '@ukis/services/src/app/layers/layers.service';
import { ILayer, IRasterLayer, IVectorLayer } from '@ukis/datatypes';




import * as WorldWind from '@nasaworldwind/worldwind';
import { Observable, BehaviorSubject } from 'rxjs';
import { ArgumentType } from '@angular/compiler/src/core';
import { Subject } from 'rxjs/Subject';


@Component({
  selector: 'ukis-www-map',
  templateUrl: './www-map.component.html',
  styleUrls: ['./www-map.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class WWWMapComponent implements OnInit, AfterViewInit, OnDestroy {
  @Input('store') store: any; //Array of GeoJSON fetures
  @Input('layers') layers: LayersService;

  map: WorldWind.WorldWindow;
  drawCtx: WorldWind.DrawContext;
  eyePosition: WorldWind.Position;

  eyePosCtxSubj: Observable<any>;

  zoom: number; //default value
  center: any //default value
  mapState: MapState;


  mapStateOn: Subscription;
  ctxStateOn: Subscription;

  layersAddRemoveOn: Subscription;
  layersPropOn: Subscription;
  baselayersAddRemoveOn: Subscription;
  baselayersPropOn: Subscription;
  mapOn: Function;


  constructor(@Inject(MapstateService) private mapStateSvc: MapstateService, @Inject(WWWMapSvc) private mapSvc: WWWMapSvc) {
    this.zoom = 3;
    this.center = {
      lat: 48.085,
      lon: 11.277
    };


    let ms = new MapState(this.zoom, this.center);
    this.mapState = ms;
    this.mapStateSvc.setMapState(ms);
    ///define initial center and zoom
  }

  ngOnInit() {
    this.initMap();
  }

  ngAfterViewInit() {
    this.map.redraw();
  }

  ngOnDestroy() {
    this.mapStateOn.unsubscribe();
    this.layersAddRemoveOn.unsubscribe();
    this.layersPropOn.unsubscribe();
    this.baselayersAddRemoveOn.unsubscribe();
    this.baselayersPropOn.unsubscribe();
    this.map.un('moveend', this.mapOn)
  }

  private initMap() {
    this.map = new WorldWind.WorldWindow("mapCanvas");
    this.mapSvc.map = this.map;
    this.drawCtx = this.map.drawContext;
    this.eyePosition = this.map.drawContext.eyePosition;
    this.eyePosCtxSubj = new Observable(this.eyePosition);

    // Add some image layers to the WorldWindow's globe.
    this.map.addLayer(new WorldWind.BMNGOneImageLayer());
    //this.map.addLayer(new WorldWind.BingAerialWithLabelsLayer(null));
   // this.map.addLayer(new WorldWind.OpenStreetMapImageLayer(null));

    // Add a compass, a coordinates display and some view controls to the WorldWindow.
    // this.map.addLayer(new WorldWind.CompassLayer());
    this.map.addLayer(new WorldWind.CoordinatesDisplayLayer(this.map));
    this.map.addLayer(new WorldWind.ViewControlsLayer(this.map));
    this.map.addLayer(new WorldWind.StarFieldLayer());
    this.map.addLayer(new WorldWind.AtmosphereLayer());


   // this.map.goTo(new WorldWind.Position(this.mapState.center.lat, this.mapState.center.lon, 60000000/(Math.pow(this.mapState.zoom, 3))));

    this.map.navigator.lookAtLocation.latitude = this.mapState.center.lat;
    this.map.navigator.lookAtLocation.longitude =this.mapState.center.lon;
    this.map.navigator.range = 60000000/(Math.pow(this.mapState.zoom, 3));

// Redraw the WorldWindow.
    this.map.redraw();




    //----------------------
    //add/remove layers
    this.baselayersAddRemoveOn = this.layers.getBaseLayers().subscribe(layers => {
      if (layers.length != this.mapSvc.getLayers('baselayers').length) {
        this.mapSvc.setLayers('baselayers', layers)
      }
    });

    //change layer visibility
    this.baselayersPropOn = this.layers.getBaseLayers().flatMap(layers => layers).subscribe(layer => {
      let bllayer = this.mapSvc.getLayerByKey({ key: 'ukisID', value: layer.ukisID }, 'baselayers');
     // console.log(bllayer)
      if (bllayer) {
        bllayer.setVisible(layer.visible);
      }
    });


    //add/remove layers
    this.layersAddRemoveOn = this.layers.getOverlays().subscribe(layers => {
      if (layers.length != this.mapSvc.getLayers('overlays').length) {
        this.mapSvc.setLayers('overlays', layers)
      }

    });

    //change layer visibility and opacity
    this.layersPropOn = this.layers.getOverlays().flatMap(layers => layers).subscribe(layer => {
      let ollayer = this.mapSvc.getLayerByKey({ key: 'ukisID', value: layer.ukisID }, 'overlays');
      if (ollayer) {
        ollayer.setVisible(layer.visible);
        ollayer.setOpacity(layer.opacity);
      }
    });


    this.mapStateOn = this.mapStateSvc.getMapState().subscribe(mapState => {
      this.mapState = mapState;

      if (this.mapState.options.notifier != "map"){
        this.map.goTo(new WorldWind.Position(this.mapState.center.lat, this.mapState.center.lon, 60000000/(Math.pow(this.mapState.zoom, 3))));
      }
    });


    this.mapOn = (evt) => {

      //console.log(this.eyePosition);
      let zoom = Math.round(Math.sqrt(60000000/this.drawCtx.eyePosition.altitude));
      let center = {
        lat: this.drawCtx.eyePosition.latitude,
        lon: this.drawCtx.eyePosition.longitude
      };
      let ms = new MapState(zoom, center, { notifier: 'map' });
      this.mapState = ms;
      this.mapStateSvc.setMapState(ms);
    }
    this.map.addEventListener('mousemove', this.mapOn);
    this.map.addEventListener('mouseup', this.mapOn);
    this.map.addEventListener('mousedown', this.mapOn);
    this.map.addEventListener('wheel', this.mapOn);

  }


}
