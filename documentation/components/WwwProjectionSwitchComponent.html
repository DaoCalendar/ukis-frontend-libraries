import {Component, Inject, OnInit} from '@angular/core';
import { WWWMapSvc, IWWWMapSvc } from '../www-map.service';
import * as WorldWind from '@nasaworldwind/worldwind';

import 'clarity-icons/shapes/technology-shapes';
import 'clarity-icons/shapes/travel-shapes';
import 'clarity-icons/shapes/core-shapes';
import 'clarity-icons/shapes/essential-shapes';
import {Subscription} from 'rxjs/Subscription';

@Component({
  selector: 'ukis-www-projection-switch',
  templateUrl: './www-projection-switch.component.html',
  styleUrls: ['./www-projection-switch.component.scss']
})
export class WwwProjectionSwitchComponent implements OnInit {

  projManager: any;
  map: WorldWind.WorldWindow;
  roundGlobe: WorldWind.Globe;
  mapSubscription: Subscription;
  flatGlobe: WorldWind.Globe2D;
  projectionNames = [
    "3D",
    "Equirectangular",
  //  "Mercator",
    "North Polar",
    "South Polar",
   // "North UPS",
 //   "South UPS",
  //  "North Gnomonic",
  //  "South Gnomonic"
  ];
  currentProj: any = this.projectionNames[0];

  constructor(@Inject(WWWMapSvc) private mapSvc: WWWMapSvc) {

  }

  ngOnInit() {
    this.initPrjSwitch();
  }

  initPrjSwitch() {
    this.map = this.mapSvc.map;
  }

  setProjection(projectionName) {
    console.log(projectionName);
    if (projectionName === "3D") {
      if (!this.roundGlobe) {
        this.roundGlobe = new WorldWind.Globe(new WorldWind.EarthElevationModel());
      }

      if (this.map.globe !== this.roundGlobe) {
        this.map.globe = this.roundGlobe;
        this.map.layers[0].enabled = true;
      }
    } else {
      if (!this.flatGlobe) {
        this.flatGlobe = new WorldWind.Globe2D();
      }

      if (projectionName === "Equirectangular") {
        this.flatGlobe.projection = new WorldWind.ProjectionEquirectangular();
      } else if (projectionName === "Mercator") {
        this.flatGlobe.projection = new WorldWind.ProjectionMercator();
      } else if (projectionName === "North Polar") {
        this.flatGlobe.projection = new WorldWind.ProjectionPolarEquidistant("North");
      } else if (projectionName === "South Polar") {
        this.flatGlobe.projection = new WorldWind.ProjectionPolarEquidistant("South");
      } else if (projectionName === "North UPS") {
        this.flatGlobe.projection = new WorldWind.ProjectionUPS("North");
      } else if (projectionName === "South UPS") {
        this.flatGlobe.projection = new WorldWind.ProjectionUPS("South");
      } else if (projectionName === "North Gnomonic") {
        this.flatGlobe.projection = new WorldWind.ProjectionGnomonic("North");
      } else if (projectionName === "South Gnomonic") {
        this.flatGlobe.projection = new WorldWind.ProjectionGnomonic("South");
      }

      if (this.map.globe !== this.flatGlobe) {
        this.map.globe = this.flatGlobe;
        this.map.layers[0].enabled = false;
      }
    }

    this.map.redraw();
  }

}
