import * as WorldWind from '@nasaworldwind/worldwind';
import {IRasterLayer} from '@ukis/datatypes/Layer';

export class UkisWMSLayer extends WorldWind.TiledImageLayer {

  displayName = 'WMS Layer';
  imageSize = 256;
  imageFormat = "image/png";
  pickEnabled = false;
  destCanvas: any;
  destContext: any;
  urlBuilder: any;
  attribution: any;
  topLevelTiles: any;
  enabled = false;
  crossOrigin = 'anonymous';
  version = '1.3.0';
  ukisID: any;
  opacity: number;
  cachePath: any;
  timeString: any;
  layerNames: any;
  coordinateSystem;
  zIndex: number;
  detailControl = 1.2;

  constructor(l: IRasterLayer) {

    super(new WorldWind.Sector(-85.05, 85.05, -180, 180), new WorldWind.Location(85.05, 180), 24, "image/png", l.name, 256, 256);

    this.displayName = l.displayName;
    if (l['params']['VERSION']) {
      this.version = l['params']['VERSION'];
    }

    if (l.time) {
      this.timeString = l.time;
    }

    if (l.zIndex) {
      this.zIndex=l.zIndex;
    }


    if (l['params']['VERSION'] == '1.3.0') {
      this.coordinateSystem = l['params']['CRS'];
    } else {
      this.coordinateSystem = l['params']['SRS']
    }

    this.layerNames = l['params']['LAYERS'];

    this.cachePath = l.url + this.layerNames;
    if (this.timeString) {
      this.cachePath = this.cachePath + this.timeString;
    }

    this.pickEnabled = false;
    // Create a canvas we can use when unprojecting retrieved images.
    this.destCanvas = document.createElement("canvas");
    this.destContext = this.destCanvas.getContext("2d");
    this.urlBuilder = new WorldWind.WmsUrlBuilder(l.url, this.layerNames, "", this.version, this.timeString);

    if (this.coordinateSystem) {
      this.urlBuilder.crs = this.coordinateSystem;
    }

    this.enabled = l.visible;

    this.ukisID = l.ukisID;

    this.attribution = l.attribution;

  }

  parseTimeDimensions = function (wmsLayerCapabilities) {
    var dimensions = wmsLayerCapabilities.extents || wmsLayerCapabilities.dimensions,
      parsedDimensions = null;
    if (dimensions) {
      parsedDimensions = [];
      for (var i = 0; i < dimensions.length; i++) {
        var dimension = dimensions[i];
        if (dimension.name.toLowerCase() === "time" &&
          (!dimension.units || dimension.units.toLowerCase() === "iso8601")) {
          var individualDimensions = dimension.content.split(",");
          for (var j = 0; j < individualDimensions.length; j++) {
            var individualDimension = individualDimensions[j],
              splitDimension = individualDimension.split("/");
            if (splitDimension.length === 1) {
              parsedDimensions.push(new Date(individualDimension));
            } else if (splitDimension.length === 3) {
              parsedDimensions.push(new WorldWind.PeriodicTimeSequence(individualDimension));
            }
          }
        }
      }
    }
    return parsedDimensions;
  };

  /*createTopLevelTiles = function (dc) {

    this.topLevelTiles.push(this.createTile(null, this.levels.firstLevel(), 1, 0));
    this.topLevelTiles.push(this.createTile(null, this.levels.firstLevel(), 1, 1));
  }; */

  mapSizeForLevel = function (levelNumber) {
    return 256 << (levelNumber + 1);
  };

  setVisible(value) {
    this.enabled = value;
  }

  setOpacity(value: number) {
    this.opacity = value;
  }

};
