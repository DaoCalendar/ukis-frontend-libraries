export class Layer {
  displayName: string;
  name: string;
  ukisID: string;
  extent: {
    minx: number;
    miny: number;
    maxx: number;
    maxy: number
  };

  url?: string;
  subdomains?: Array<string>;

  minResolution: number;
  maxResolution: number;
  type: string;

  opacity: number;
  visible: boolean;
  zIndex: number;

  transparent: boolean;
  attribution: string;
  time: string;
  version: string;
  continuousWorld: boolean;
  crs: string;
  legendImg: string;

  constructor(options) {
    Object.assign(this, options);
  }
}

export class RasterLayer extends Layer {
  format: string; //"image/png"
  params?: any;

  constructor(options) {
    super(options);
  }
}

export class VectorLayer extends Layer {
  data: any
  style?: Function

  constructor(options) {
    super(options);
  }
}

export interface ILayer {
  name: string
  ukisID: string
  extent: {
    minx: number
    miny: number
    maxx: number
    maxy: number
  }

  url?: string;
  subdomains?: Array<string>;

  minResolution: number
  maxResolution: number
  type: "wms" | "xyz" | "geojson"

  opacity: number
  visible: boolean
  zIndex: number
  transparent: boolean
  attribution: string
  time: string
  version: string
  continuousWorld: boolean
  crs: string
  legendImg: string
}

export interface IRasterLayer extends ILayer {
  format: string,//"image/png"
  params: any;
}

export interface IVectorLayer extends ILayer {
  data: any
  style?: Function
}
