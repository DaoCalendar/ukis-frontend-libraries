import * as WorldWind from '@nasaworldwind/worldwind';
import {IVectorLayer} from '@ukis/datatypes/Layer';

export class UkisGeoJsonLayer extends WorldWind.RenderableLayer {

  displayName = 'GeoJson Layer';
  enabled = true;
  opacity: number;
  maxActiveAltitude: number;
  minActiveAltitude: number;
  pickEnabled = true;
  zIndex: number;

  renderables: any;

  time: any;


  ukisID: any;

  placemarkAttributes: any;
  shpConfiguration: any;

  constructor(l: IVectorLayer) {

    super();

    if (l.zIndex) {
      this.zIndex=l.zIndex;
    }

    this.displayName = l.displayName;

    this.ukisID = l.ukisID;

    this.placemarkAttributes = new WorldWind.PlacemarkAttributes(null);
    this.placemarkAttributes.imageScale = 0.05;
    this.placemarkAttributes.imageColor = WorldWind.Color.WHITE;
    this.placemarkAttributes.labelAttributes.offset = new WorldWind.Offset(
      WorldWind.OFFSET_FRACTION, 0.5,
      WorldWind.OFFSET_FRACTION, 1.5);
    this.placemarkAttributes.imageSource = WorldWind.configuration.baseUrl + "images/white-dot.png";

    this.shpConfiguration = {};

    let polygonGeoJSON = new WorldWind.GeoJSONParser(l.url);
    polygonGeoJSON.load(null, this.shapeConfigurationCallback, this);

  }

  setVisible(value)  {
    this.enabled = value;
  }

  setOpacity(value: number) {
    this.opacity = value;
  }

  shapeConfigurationCallback(geometry, properties) {

    if (geometry.isPointType() || geometry.isMultiPointType()) {
      this.shpConfiguration = {
          attributes: new WorldWind.PlacemarkAttributes(this.placemarkAttributes)
      };

      if (properties && (properties.name || properties.Name || properties.NAME)) {
        this.shpConfiguration.name = properties.name || properties.Name || properties.NAME;
      }
      if (properties && properties.POP_MAX) {
        var population = properties.POP_MAX;
        this.shpConfiguration.attributes.imageScale = 0.01 * Math.log(population);
      }
    }
    else if (geometry.isLineStringType() || geometry.isMultiLineStringType()) {
      this.shpConfiguration  = {
        attributes: new WorldWind.ShapeAttributes(null)
      } ;
      this.shpConfiguration.attributes.drawOutline = true;
      this.shpConfiguration.attributes.outlineColor = new WorldWind.Color(
        0.1 * this.shpConfiguration.attributes.interiorColor.red,
        0.3 * this.shpConfiguration.attributes.interiorColor.green,
        0.7 * this.shpConfiguration.attributes.interiorColor.blue,
        1.0);
      this.shpConfiguration.attributes.outlineWidth = 1.0;
    }
    else if (geometry.isPolygonType() || geometry.isMultiPolygonType()) {
      this.shpConfiguration = {
        attributes: new WorldWind.ShapeAttributes(null)
      };

      // Fill the polygon with a random pastel color.
      this.shpConfiguration.attributes.interiorColor = new WorldWind.Color(
        0.375 + 0.5 * Math.random(),
        0.375 + 0.5 * Math.random(),
        0.375 + 0.5 * Math.random(),
        0.1);
      // Paint the outline in a darker variant of the interior color.
      this.shpConfiguration.attributes.outlineColor = new WorldWind.Color(
        0.5 * this.shpConfiguration.attributes.interiorColor.red,
        0.5 * this.shpConfiguration.attributes.interiorColor.green,
        0.5 * this.shpConfiguration.attributes.interiorColor.blue,
        1.0);
    }

    return this.shpConfiguration;
  };

};
