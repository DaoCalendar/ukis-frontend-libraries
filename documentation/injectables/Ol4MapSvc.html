import { Injectable } from '@angular/core';
import * as ol from 'openlayers';
import { Layer, RasterLayer, IRasterLayer, VectorLayer } from '@ukis/datatypes';
declare var proj4: any;


export interface IOl4MapSvc {
  setZoom(zoom: number);
  map: ol.Map;
  getLayers(type: 'overlays' | 'baselayers');
  addLayer(layer: any, type: 'overlays' | 'baselayers');
  addLayers(layers: Array<any>, type: 'overlays' | 'baselayers');
  addBaseLayers(layers: Array<Layer>);
  getLayerByKey(key: { key: string, value: string }, type: 'overlays' | 'baselayers');
  removeLayerByKey(key: { key: string, value: string }, type: 'overlays' | 'baselayers');
  removeAllLayers(type: 'overlays' | 'baselayers');
  /*
  * extent: ['xmin', 'ymin', 'xmax', 'ymax']
  */
  setExtent(extent: ol.Extent, geographic?: boolean, fitOptions?: any): any;
  setCenter(center: ol.Coordinate, geographic?: boolean)

  temp: any;
  EPSG: string;
}


@Injectable()
export class Ol4MapSvc implements IOl4MapSvc {
  map: any; //ol.Map;
  temp: any;
  EPSG: string;

  constructor() {
    this.map = {};
    this.temp = {};
    this.EPSG = 'EPSG:4326';//'EPSG:4326'; EPSG:3857
  }

  public getLayers(type: 'overlays' | 'baselayers') {
    var layers;
    this.map.getLayers().getArray().forEach((layerGroup) => {
      if (layerGroup.get('type') === type) {
        layers = layerGroup.getLayers().getArray();
      }
    })
    return layers;
  }

  public getLayerByKey(key: { key: string, value: string }, type: 'overlays' | 'baselayers') {
    var layers = this.getLayers(type);
    var _layer;
    layers.forEach((layer) => {
      if (layer.get(key.key) && layer.get(key.key) == key.value) {
        _layer = layer
      }
    })
    return _layer;
  }

  public addLayer(layer: any, type: 'overlays' | 'baselayers') {
    var layers;
    this.map.getLayers().getArray().forEach((layerGroup) => {
      if (layerGroup.get('type') === type) {
        layers = layerGroup.getLayers().getArray();
        layers.push(layer);
        layerGroup.setLayers(new ol.Collection(layers))
      }
    })
    return layers;
  }

  public addLayers(layers: Array<any>, type: 'overlays' | 'baselayers') {
    var _layers;
    this.map.getLayers().getArray().forEach((layerGroup: ol.layer.Group) => {
      if (layerGroup.get('type') === type) {
        /*
        _layers = layerGroup.getLayers().getArray();
        _layers.push(layer);
        */
        _layers = layers;
        layerGroup.setLayers(new ol.Collection(_layers))
      }
    })
    return layers;
  }

  public removeLayerByKey(key: { key: string, value: string }, type: 'overlays' | 'baselayers') {
    var layer = this.getLayerByKey(key, type);
    this.map.removeLayer(layer);
  }

  public removeAllLayers(type: 'overlays' | 'baselayers') {
    var layers;
    this.map.getLayers().getArray().forEach((layerGroup) => {
      if (layerGroup.get('type') === type) {
        layers = layerGroup.getLayers()
        layers.clear();
      }
    })

  }

  public setExtent(extent: ol.Extent, geographic?: boolean, fitOptions?: any): any {
    //var _extent = ol.extent.boundingExtent([destLoc,currentLoc]);
    var projection = (geographic) ? ol.proj.get('EPSG:4326') : ol.proj.get(this.EPSG);
    var transfomExtent = ol.proj.transformExtent(extent, projection, this.map.getView().getProjection().getCode());
    var _fitOptions = {
      size: this.map.getSize(),
      padding: [100, 200, 100, 100] //Padding (in pixels) to be cleared inside the view. Values in the array are top, right, bottom and left padding. Default is [0, 0, 0, 0].
    }
    if (fitOptions) {
      Object.assign(_fitOptions, fitOptions)
    }
    this.map.getView().fit(transfomExtent, fitOptions);
    return transfomExtent;
  }

    console.log('set center in svc', transfomCenter)

    this.map.getView().setCenter(transfomCenter);
  }

  public geoJsonToFeature(geojson: any): any {
    var GEOJSON = new ol.format.GeoJSON({
      defaultDataProjection: 'EPSG:4326',
      featureProjection: this.EPSG
    });
    return GEOJSON.readFeature(geojson)
  }

  public geoJsonToFeatures(geojson: any): Array<any> {
    var GEOJSON = new ol.format.GeoJSON({
      defaultDataProjection: 'EPSG:4326',
      featureProjection: this.EPSG
    });
    return GEOJSON.readFeatures(geojson)
  }
